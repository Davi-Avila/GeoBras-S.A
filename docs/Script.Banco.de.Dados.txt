CREATE TABLE Cliente(
	ID_CLIENTE INT PRIMARY KEY,
    NOME_CLIENTE VARCHAR(100),
    EMAIL VARCHAR(200),
    PAIS VARCHAR(50),
    CEP VARCHAR(9),
    CPF VARCHAR(14),
    CNPJ VARCHAR(18)
    );


CREATE TABLE ORCAMENTO(
	ID_ORCAMENTO INT PRIMARY KEY,
    DESLOCAMENTO DECIMAL(16,2),
    MAO_DE_OBRA DECIMAL(16,2),
	SERVICO VARCHAR(200),
    ALUQUEL_DE_EQUIPAMENTO DECIMAL(16,2),
    IMPOSTO DECIMAL(16,2),
    ORCAMENTO_TOTAL DECIMAL(16,2)
);
CREATE TABLE MATERIAL(
	ID_MATERIAL INT PRIMARY KEY,
    NOME_MATERIAL VARCHAR(100),
    VALOR_MATERIAL DECIMAL(16,2),
    QUANTIDADE INT,
    ID_ORCAMENTO INT UNIQUE,
    FOREIGN KEY (ID_ORCAMENTO) REFERENCES ORCAMENTO(ID_ORCAMENTO)
);
CREATE TABLE OBRA(
	ID_OBRA INT PRIMARY KEY,
    NOME_OBRA VARCHAR(100),
    ENDERECO VARCHAR(100),
    DATA_INICIO DATE,
    DATA_FIM_PREVISTO DATE,
    ETAPA VARCHAR(200),
    ID_ORCAMENTO INT UNIQUE,
    ID_CLIENTE INT UNIQUE,
    FOREIGN KEY (ID_ORCAMENTO) REFERENCES ORCAMENTO(ID_ORCAMENTO),
    FOREIGN KEY (ID_CLIENTE) REFERENCES CLIENTE(ID_CLIENTE)
);
