CREATE TABLE ORCAMENTO(
	ID_ORCAMENTO INT PRIMARY KEY,
    NOME_ORCAMENTO VARCHAR(100),
    DESLOCAMENTO DECIMAL(16,2),
    MAO_DE_OBRA DECIMAL(16,2),
    SERVICO VARCHAR(100),
    ALUGUEL_DE_EQUIPAMENTO DECIMAL(16,2),
    SOMA_MATERIAIS DECIMAL(16,2),
    IMPOSTO DECIMAL(16,2),
    ORCAMENTO_TOTAL DECIMAL(16,2)
    );
CREATE TABLE CLIENTE(
	ID_CLIENTE INT PRIMARY KEY,
    NOME_CLIENTE VARCHAR(100) NOT NULL,
    EMAIL VARCHAR(100) NOT NULL,
    PAIS VARCHAR(100) NOT NULL,
    CEP VARCHAR(9) NOT NULL,
    CPF VARCHAR(0) NOT NULL,
    CNPJ VARCHAR(100)
    );

CREATE TABLE OBRA(
	ID_OBRA INT PRIMARY KEY,
	NOME_OBRA VARCHAR(100) NOT NULL,
    ENDERECO VARCHAR(100) NOT NULL,
    DATA_INICIO DATE NOT NULL,
    DATA_FIM DATE NOT NULL,
    ETAPA VARCHAR(100),
    ID_CLIENTE INT UNIQUE NOT NULL,
    ID_ORCAMENTO INT UNIQUE NOT NULL,
    
    FOREIGN KEY (ID_CLIENTE) REFERENCES CLIENTE(ID_CLIENTE),
    FOREIGN KEY (ID_ORCAMENTO) REFERENCES ORCAMENTO(ID_ORCAMENTO)
    );
